allGenre = list(genreList.columns.unique())
#Declaring function to get the best genre from dataframe input
def getGenre(df):
    max = 0
    for genre in allGenre:
        if df[str(genre)].sum()>max:
            best = genre
            max = df[str(genre)].sum()
    return best
